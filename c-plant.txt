### 3. storage.cpp - Implementation  
```cpp
#include "storage.hpp"
#include "constants.hpp"
#include <fstream>
#include <iostream>

namespace cmdvault::storage {

std::filesystem::path get_file_path() {
    auto home = std::getenv("HOME");
    if (!home) {
        return std::filesystem::current_path() / VAULT_FILENAME;
    }
    return std::filesystem::path(home) / VAULT_FILENAME;
}

std::optional<std::vector<CommandItem>> load_items() {
    auto path = get_file_path();
    if (!std::filesystem::exists(path)) {
        return std::nullopt;
    }
    
    std::ifstream file(path);
    if (!file.is_open()) {
        return std::nullopt;
    }
    
    try {
        nlohmann::json j;
        file >> j;
        return j.get<std::vector<CommandItem>>();
    } catch (const std::exception& e) {
        std::cerr << "Error loading commands: " << e.what() << std::endl;
        return std::nullopt;
    }
}

bool save_items(const std::vector<CommandItem>& items) {
    auto path = get_file_path();
    std::ofstream file(path);
    if (!file.is_open()) {
        return false;
    }
    
    try {
        nlohmann::json j = items;
        file << j.dump(2);  // Pretty print with 2 spaces
        return true;
    } catch (const std::exception& e) {
        std::cerr << "Error saving commands: " << e.what() << std::endl;
        return false;
    }
}

} // namespace cmdvault::storage
```

#include "cli.hpp"
#include "ui.hpp"
#include "app.hpp"
#include <iostream>
#include <exception>

int main(int argc, char* argv[]) {
    try {
        auto args = cmdvault::cli::parse_args(argc, argv);
        
        if (args.launch_tui) {
            cmdvault::App app;
            cmdvault::ui::UI ui(app);
            ui.run();
        } else {
            cmdvault::cli::handle_cli_commands(args);
        }
        
        return 0;
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
}

## 🔨 CMakeLists.txt Configuration

```cmake
cmake_minimum_required(VERSION 3.16)
project(cmd-vault-cpp VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find dependencies
find_package(PkgConfig REQUIRED)
find_package(nlohmann_json REQUIRED)
find_package(cxxopts REQUIRED)
find_package(ftxui REQUIRED)
find_package(Threads REQUIRED)

# Source files
set(SOURCES
    src/main.cpp
    src/app.cpp
    src/cli.cpp  
    src/handler.cpp
    src/models.cpp
    src/placeholders.cpp
    src/storage.cpp
    src/ui.cpp
    src/utils.cpp
)

# Create executable
add_executable(cmd-vault ${SOURCES})

# Include directories
target_include_directories(cmd-vault PRIVATE include)

# Link libraries
target_link_libraries(cmd-vault 
    nlohmann_json::nlohmann_json
    cxxopts::cxxopts
    ftxui::screen
    ftxui::dom  
    ftxui::component
    Threads::Threads
)

# Compiler flags
target_compile_options(cmd-vault PRIVATE
    -Wall -Wextra -Wpedantic
    $<$<CONFIG:Release>:-O3>
    $<$<CONFIG:Debug>:-g -fsanitize=address>
)

# Install target
install(TARGETS cmd-vault DESTINATION bin)
```

## 🛠️ Key Implementation Differences from Rust

### Memory Management
- Use smart pointers (`std::unique_ptr`, `std::shared_ptr`) instead of Rust's ownership
- RAII for resource management
- Optional type (`std::optional`) for nullable values

### Error Handling  
- Use exceptions for error propagation instead of `Result<T, E>`
- Return `std::optional` for operations that might fail
- Validate inputs and provide meaningful error messages

### String Handling
- Use `std::string` and `std::string_view` for efficiency
- Be careful with string lifetime and copies
- Use `const std::string&` for parameters to avoid copies

### Collections
- `std::vector` instead of `Vec`
- `std::map`/`std::unordered_map` for key-value storage
- Use algorithms from `<algorithm>` for operations like `sort`, `find_if`

### Async/Threading
- Use `std::thread` and `std::async` if needed
- Mutex (`std::mutex`) for thread safety
- Atomic operations (`std::atomic`) for simple shared state

## 📝 Additional Notes

### Performance Optimizations
- Use `std::string_view` for read-only string parameters
- Reserve vector capacity when size is known
- Use move semantics (`std::move`) to avoid unnecessary copies
- Consider using `std::span` for array-like parameters (C++20)

### Code Quality
- Use static analysis tools (`clang-tidy`, `cppcheck`)
- Enable all compiler warnings (`-Wall -Wextra -Wpedantic`)
- Use address sanitizer in debug builds (`-fsanitize=address`)
- Follow modern C++ guidelines (C++ Core Guidelines)

### Cross-Platform Compatibility
- Test on Windows, macOS, and Linux
- Use CMake for portable build configuration
- Avoid platform-specific APIs where possible
- Use feature detection rather than OS detection

This implementation plan provides a complete roadmap for recreating the Rust cmd-vault in modern C++ while maintaining all the original functionality and features.
